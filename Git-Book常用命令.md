## 查看历史版本
gitbook ls-remote
## 安装历史版本
gitbook fetch beta
## 卸载对应版本
gitbook uninstall 2.0.1
## 卸载gitbook
npm uninstall -g gitbook
## 生成静态页面
gitbook build
## 初始化 gitbook
gitbook init
## 生成静态页面并运行服务器
gitbook serve
## 查看所有命令
gitbook help
## 更新gitbook到最新版本
gitbook update
## 输出错误信息
gitbook build --debug
